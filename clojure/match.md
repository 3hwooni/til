```clojure
(defn spread
  ([x] x)
  ([x y] (- (max x y) (min x y)))
  ([x y z] (- (max x y z) (min x y z))))
```

```clojure
(defn spread [& nums]
  (- (apply max nums) (apply min nums)))
```

```clojure
(defn product
  ([a] a)
  ([a b] (reduce #(concat %1 (map (partial vector %2) b)) [] a))
  ([a b & seq]
   (let [p (apply product (cons b seq))]
     (reduce #(concat %1 (map (partial cons %2) p)) [] a))))
```
